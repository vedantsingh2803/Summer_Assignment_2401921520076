class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int bestSum = nums[0];
        int currentSum = nums[0];

        for (int i = 1; i < nums.size(); i++) {
            if (currentSum < 0) {
                currentSum = nums[i];
            } else {
                currentSum += nums[i];
            }

            if (currentSum > bestSum) {
                bestSum = currentSum;
            }
        }

        return bestSum;
    }
};
